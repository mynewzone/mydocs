
//重启nginx
1.检查nginx配置文件，进入nginx安装目录sbin下，输入命令./nginx -t
2.重启：进入nginx可执行目录sbin下，输入命令./nginx -s reload 即可

//nginx配多个域名
server_name  yaoshibang.cn www.yaoshibang.cn;
